List<Page>? parsePages(dynamic data) {
  if (data is String) {
    return jsonDecode(data)
        .cast<Map<String, dynamic>>()
        .map<Page>((json) => Page.fromMap(json))
        .toList();
  } else {
    return data
        .cast<Map<String, dynamic>>()
        .map<Page>((json) => Page.fromMap(json))
        .toList();
  }
}
